import { supabase } from '../lib/supabase';
import { apiService, ApiError, ApiErrorType } from './apiService';
import type { User } from '@supabase/supabase-js';
 
// Interface para permissões
export interface Permission {
  resource: string;
  action: 'read' | 'write' | 'delete' | 'admin';
  conditions?: Record<string, any>;
}
 
// Interface para contexto de autenticação
export interface AuthContext {
  user: User | null;
  session: any;
  permissions: Permission[];
  isLoading: boolean;
  isAuthenticated: boolean;
}
 
// Interface para middleware options
export interface MiddlewareOptions {
  requireAuth?: boolean;
  requiredPermissions?: Permission[];
  allowedRoles?: string[];
  redirectTo?: string;
  onAuthError?: (error: ApiError) => void;
}
 
// Cache de permissões em memória
const permissionCache = new Map<string, { permissions: Permission[]; timestamp: number }>();
const PERMISSION_CACHE_TTL = 5 * 60 * 1000; // 5 minutos
 
// Mapeamento de papéis para permissões padrão
const ROLE_PERMISSIONS: Record<string, Permission[]> = {
  admin: [
    { resource: '*', action: 'admin' }
  ],
  manager: [
    { resource: 'actions', action: 'read' },
    { resource: 'actions', action: 'write' },
    { resource: 'risks', action: 'read' },
    { resource: 'risks', action: 'write' },
    { resource: 'reports', action: 'read' },
    { resource: 'users', action: 'read' }
  ],
  analyst: [
    { resource: 'actions', action: 'read' },
    { resource: 'risks', action: 'read' },
    { resource: 'reports', action: 'read' }
  ],
  viewer: [
    { resource: 'actions', action: 'read' },
    { resource: 'risks', action: 'read' }
  ]
};
 
// Classe de middleware de autenticação
export class AuthMiddleware {
  private authContext: AuthContext = {
    user: null,
    session: null,
    permissions: [],
    isLoading: true,
    isAuthenticated: false
  };
 
  private authStateListeners: Set<(context: AuthContext) => void> = new Set();
 
  constructor() {
    this.initializeAuth();
  }
 
  // Inicializar estado de autenticação
  private async initializeAuth(): Promise<void> {
    try {
      const { data: { session } } = await supabase.auth.getSession();
 
      if (session?.user) {
        await this.updateAuthContext(session.user, session);
      } else {
        this.setAuthContext({
          user: null,
          session: null,
          permissions: [],
          isLoading: false,
          isAuthenticated: false
        });
      }
 
      // Configurar listener para mudanças de autenticação
      supabase.auth.onAuthStateChange(async (event, session) => {
        console.log('Auth state change:', event);
 
        if (event === 'SIGNED_IN' && session?.user) {
          await this.updateAuthContext(session.user, session);
        } else if (event === 'SIGNED_OUT') {
          this.setAuthContext({
            user: null,
            session: null,
            permissions: [],
            isLoading: false,
            isAuthenticated: false
          });
        } else if (event === 'TOKEN_REFRESHED' && session?.user) {
          await this.updateAuthContext(session.user, session);
        }
      });
    } catch (error) {
      console.error('Error initializing auth:', error);
      this.setAuthContext({
        user: null,
        session: null,
        permissions: [],
        isLoading: false,
        isAuthenticated: false
      });
    }
  }
 
  // Atualizar contexto de autenticação
  private async updateAuthContext(user: User, session: any): Promise<void> {
    try {
      this.setAuthContext(prev => ({
        ...prev,
        isLoading: true
      }));
 
      // Buscar permissões do usuário
      const permissions = await this.fetchUserPermissions(user);
 
      this.setAuthContext({
        user,
        session,
        permissions,
        isLoading: false,
        isAuthenticated: true
      });
 
    } catch (error) {
      console.error('Error updating auth context:', error);
      this.setAuthContext(prev => ({
        ...prev,
        isLoading: false,
        isAuthenticated: false
      }));
    }
  }
 
  // Buscar permissões do usuário
  private async fetchUserPermissions(user: User): Promise<Permission[]> {
    try {
      // Verificar cache primeiro
      const cacheKey = user.id;
      const cached = permissionCache.get(cacheKey);
 
      if (cached && Date.now() - cached.timestamp < PERMISSION_CACHE_TTL) {
        return cached.permissions;
      }
 
      // Buscar permissões do banco de dados
      const { data: userPermissions, error } = await supabase
        .from('user_permissions')
        .select('resource, action, conditions')
        .eq('user_id', user.id);
 
      if (error) throw error;
 
      // Buscar role do usuário
      const { data: profile, error: profileError } = await supabase
        .from('profiles')
        .select('role')
        .eq('id', user.id)
        .single();
 
      if (profileError && profileError.code !== 'PGRST116') {
        throw profileError;
      }
 
      // Combinar permissões específicas com permissões do papel
      const role = profile?.role || 'viewer';
      const rolePermissions = ROLE_PERMISSIONS[role] || ROLE_PERMISSIONS.viewer;
      const specificPermissions = userPermissions || [];
 
      const allPermissions = [
        ...rolePermissions,
        ...specificPermissions.map((p: any) => ({
          resource: p.resource,
          action: p.action,
          conditions: p.conditions
        }))
      ];
 
      // Salvar no cache
      permissionCache.set(cacheKey, {
        permissions: allPermissions,
        timestamp: Date.now()
      });
 
      return allPermissions;
 
    } catch (error) {
      console.error('Error fetching user permissions:', error);
      // Retornar permissões básicas em caso de erro
      return ROLE_PERMISSIONS.viewer;
    }
  }
 
  // Atualizar contexto de autenticação (thread-safe)
  private setAuthContext(updater: AuthContext | ((prev: AuthContext) => AuthContext)): void {
    const newContext = typeof updater === 'function' ? updater(this.authContext) : updater;
    this.authContext = newContext;
 
    // Notificar listeners
    this.authStateListeners.forEach(listener => {
      try {
        listener(newContext);
      } catch (error) {
        console.error('Error in auth state listener:', error);
      }
    });
  }
 
  // Middleware principal
  public async authenticate(options: MiddlewareOptions = {}): Promise<AuthContext> {
    const {
      requireAuth = true,
      requiredPermissions = [],
      allowedRoles = [],
      redirectTo = '/login',
      onAuthError
    } = options;
 
    try {
      // Aguardar carregamento inicial
      if (this.authContext.isLoading) {
        await this.waitForAuthLoad();
      }
 
      // Verificar autenticação obrigatória
      if (requireAuth && !this.authContext.isAuthenticated) {
        throw new ApiError(
          'Autenticação necessária',
          ApiErrorType.AUTHENTICATION_ERROR,
          'AUTH_REQUIRED'
        );
      }
 
      // Verificar permissões necessárias
      if (requiredPermissions.length > 0) {
        const hasAllPermissions = requiredPermissions.every(requiredPermission =>
          this.hasPermission(requiredPermission)
        );
 
        if (!hasAllPermissions) {
          throw new ApiError(
            'Permissões insuficientes',
            ApiErrorType.AUTHORIZATION_ERROR,
            'INSUFFICIENT_PERMISSIONS'
          );
        }
      }
 
      // Verificar papéis permitidos
      if (allowedRoles.length > 0 && this.authContext.user) {
        const userRole = this.authContext.user.user_metadata?.role ||
                         this.authContext.user.app_metadata?.role ||
                         'viewer';
 
        if (!allowedRoles.includes(userRole)) {
          throw new ApiError(
            'Papel não autorizado',
            ApiErrorType.AUTHORIZATION_ERROR,
            'UNAUTHORIZED_ROLE'
          );
        }
      }
 
      return this.authContext;
 
    } catch (error) {
      if (error instanceof ApiError) {
        if (onAuthError) {
          onAuthError(error);
        }
 
        // Redirecionar para login em caso de erro de autenticação
        if (error.type === ApiErrorType.AUTHENTICATION_ERROR && redirectTo) {
          if (typeof window !== 'undefined') {
            window.location.href = redirectTo;
          }
        }
      }
      throw error;
    }
  }
 
  // Verificar permissão específica
  public hasPermission(permission: Permission): boolean {
    return this.authContext.permissions.some(p =>
      p.resource === permission.resource &&
      p.action === permission.action &&
      this.checkConditions(p.conditions, permission.conditions)
    );
  }
 
  // Verificar condições de permissão
  private checkConditions(
    storedConditions: Record<string, any> | undefined,
    requiredConditions: Record<string, any> | undefined
  ): boolean {
    if (!requiredConditions) return true;
    if (!storedConditions) return false;
 
    return Object.entries(requiredConditions).every(([key, value]) => {
      return storedConditions[key] === value;
    });
  }
 
  // Aguardar carregamento de autenticação
  private async waitForAuthLoad(): Promise<void> {
    return new Promise((resolve) => {
      const checkInterval = setInterval(() => {
        if (!this.authContext.isLoading) {
          clearInterval(checkInterval);
          resolve();
        }
      }, 100);
    });
  }
 
  // Obter contexto atual
  public getAuthContext(): AuthContext {
    return { ...this.authContext };
  }
 
  // Inscrever para mudanças de estado
  public onAuthStateChange(listener: (context: AuthContext) => void): () => void {
    this.authStateListeners.add(listener);
 
    // Retornar função de unsubscribe
    return () => {
      this.authStateListeners.delete(listener);
    };
  }
 
  // Forçar refresh de autenticação
  public async refreshAuth(): Promise<void> {
    try {
      const { data: { session } } = await supabase.auth.getSession();
 
      if (session?.user) {
        await this.updateAuthContext(session.user, session);
      } else {
        this.setAuthContext({
          user: null,
          session: null,
          permissions: [],
          isLoading: false,
          isAuthenticated: false
        });
      }
    } catch (error) {
      console.error('Error refreshing auth:', error);
    }
  }
 
  // Limpar cache de permissões
  public clearPermissionCache(): void {
    permissionCache.clear();
  }
 
  // Verificar se usuário tem papel específico
  public hasRole(role: string): boolean {
    if (!this.authContext.user) return false;
 
    const userRole = this.authContext.user.user_metadata?.role ||
                     this.authContext.user.app_metadata?.role ||
                     'viewer';
 
    return userRole === role;
  }
 
  // Verificar se usuário é administrador
  public isAdmin(): boolean {
    return this.hasRole('admin');
  }
 
  // Obter informações do usuário atual
  public getCurrentUser(): User | null {
    return this.authContext.user;
  }
 
  // Obter ID do usuário atual
  public getCurrentUserId(): string | null {
    return this.authContext.user?.id || null;
  }
 
  // Verificar se usuário está autenticado
  public isAuthenticated(): boolean {
    return this.authContext.isAuthenticated && !this.authContext.isLoading;
  }
}
 
// Hook personalizado para usar o middleware (para React)
export function useAuthMiddleware() {
  // Esta função pode ser usada como base para um hook React
  // mas por enquanto retorna a instância estática
  return authMiddleware;
}
 
// Instância estática para uso global
export const authMiddleware = new AuthMiddleware();
 
// Exportar funções de conveniência
export const requireAuth = (options: Omit<MiddlewareOptions, 'requireAuth'> = {}) =>
  authMiddleware.authenticate({ ...options, requireAuth: true });
 
export const requirePermission = (permission: Permission, options: Omit<MiddlewareOptions, 'requireAuth' | 'requiredPermissions'> = {}) =>
  authMiddleware.authenticate({ ...options, requireAuth: true, requiredPermissions: [permission] });
 
export const requireRole = (role: string, options: Omit<MiddlewareOptions, 'requireAuth' | 'allowedRoles'> = {}) =>
  authMiddleware.authenticate({ ...options, requireAuth: true, allowedRoles: [role] });
