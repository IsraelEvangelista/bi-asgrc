import { ApiError, ApiErrorType } from './apiService';
 
// Interface para configuração CORS
export interface CORSConfig {
  allowedOrigins: string[];
  allowedMethods: string[];
  allowedHeaders: string[];
  exposedHeaders: string[];
  maxAge: number;
  credentials: boolean;
  preflightContinue: boolean;
  optionsSuccessStatus: number;
}
 
// Interface para headers CORS
export interface CORSHeaders {
  'Access-Control-Allow-Origin': string;
  'Access-Control-Allow-Methods': string;
  'Access-Control-Allow-Headers': string;
  'Access-Control-Max-Age': string;
  'Access-Control-Allow-Credentials': string;
  'Access-Control-Expose-Headers'?: string;
}
 
// Interface para requisição simulada (para ambiente frontend)
export interface RequestInfo {
  origin?: string;
  method?: string;
  headers?: Record<string, string>;
}
 
// Configuração padrão de segurança
const DEFAULT_CORS_CONFIG: CORSConfig = {
  allowedOrigins: [], // Vazio significa "allow all" para desenvolvimento
  allowedMethods: ['GET', 'POST', 'PUT', 'DELETE', 'PATCH', 'OPTIONS'],
  allowedHeaders: [
    'Content-Type',
    'Authorization',
    'X-Requested-With',
    'X-Api-Key',
    'X-CSRF-Token',
    'Accept',
    'Cache-Control',
    'Pragma'
  ],
  exposedHeaders: [
    'X-Request-ID',
    'X-RateLimit-Limit',
    'X-RateLimit-Remaining',
    'X-RateLimit-Reset'
  ],
  maxAge: 86400, // 24 horas
  credentials: true,
  preflightContinue: false,
  optionsSuccessStatus: 204
};
 
// Headers de segurança padrão
const SECURITY_HEADERS = {
  'X-Content-Type-Options': 'nosniff',
  'X-Frame-Options': 'DENY',
  'X-XSS-Protection': '1; mode=block',
  'Strict-Transport-Security': 'max-age=31536000; includeSubDomains',
  'Referrer-Policy': 'strict-origin-when-cross-origin',
  'Content-Security-Policy': "default-src 'self'; script-src 'self' 'unsafe-inline' 'unsafe-eval'; style-src 'self' 'unsafe-inline'; img-src 'self' data: https:; font-src 'self' data:; connect-src 'self' https:; frame-ancestors 'none';",
  'Permissions-Policy': 'camera=(), microphone=(), geolocation=(), payment=()'
};
 
// Classe de serviço CORS
export class CORSService {
  private config: CORSConfig;
  private securityHeaders: Record<string, string>;
 
  constructor(config: Partial<CORSConfig> = {}) {
    this.config = { ...DEFAULT_CORS_CONFIG, ...config };
    this.securityHeaders = { ...SECURITY_HEADERS };
  }
 
  // Verificar se uma origem é permitida
  public isOriginAllowed(origin: string): boolean {
    // Se não houver origens configuradas, permitir todas
    if (this.config.allowedOrigins.length === 0) {
      return true;
    }
 
    // Permitir localhost em desenvolvimento
    if (origin.includes('localhost') || origin.includes('127.0.0.1')) {
      return true;
    }
 
    // Verificar se a origem está na lista de permitidas
    return this.config.allowedOrigins.some(allowedOrigin => {
      // Suporte a wildcards
      if (allowedOrigin.includes('*')) {
        const regex = new RegExp(allowedOrigin.replace(/\*/g, '.*'));
        return regex.test(origin);
      }
      return allowedOrigin === origin;
    });
  }
 
  // Gerar headers CORS para requisições simples
  public generateCORSHeaders(request: RequestInfo): CORSHeaders {
    const origin = request.origin || '';
 
    if (!this.isOriginAllowed(origin)) {
      // Se origem não permitida, retornar headers restritivos
      return {
        'Access-Control-Allow-Origin': 'null',
        'Access-Control-Allow-Methods': '',
        'Access-Control-Allow-Headers': '',
        'Access-Control-Max-Age': '0',
        'Access-Control-Allow-Credentials': 'false'
      };
    }
 
    // Configurar valor dinâmico para Allow-Origin
    const allowOrigin = this.config.allowedOrigins.length === 0 ? '*' : origin;
 
    return {
      'Access-Control-Allow-Origin': allowOrigin,
      'Access-Control-Allow-Methods': this.config.allowedMethods.join(', '),
      'Access-Control-Allow-Headers': this.config.allowedHeaders.join(', '),
      'Access-Control-Max-Age': this.config.maxAge.toString(),
      'Access-Control-Allow-Credentials': this.config.credentials.toString()
    };
  }
 
  // Gerar headers para requisições preflight
  public generatePreflightHeaders(request: RequestInfo): CORSHeaders {
    const corsHeaders = this.generateCORSHeaders(request);
 
    // Adicionar headers expostos se configurados
    if (this.config.exposedHeaders.length > 0) {
      corsHeaders['Access-Control-Expose-Headers'] = this.config.exposedHeaders.join(', ');
    }
 
    return corsHeaders;
  }
 
  // Verificar se método é permitido
  public isMethodAllowed(method: string): boolean {
    return this.config.allowedMethods.includes(method.toUpperCase());
  }
 
  // Verificar se headers são permitidos
  public areHeadersAllowed(headers: Record<string, string>): boolean {
    const requestedHeaders = Object.keys(headers).map(h => h.toLowerCase());
 
    return requestedHeaders.every(header =>
      this.config.allowedHeaders.some(allowed =>
        header.toLowerCase() === allowed.toLowerCase()
      )
    );
  }
 
  // Validar requisição CORS
  public validateCORSRequest(request: RequestInfo): {
    isValid: boolean;
    error?: string;
    headers?: Record<string, string>;
  } {
    try {
      // Verificar origem
      if (request.origin && !this.isOriginAllowed(request.origin)) {
        return {
          isValid: false,
          error: 'Origin not allowed by CORS policy'
        };
      }
 
      // Verificar método
      if (request.method && !this.isMethodAllowed(request.method)) {
        return {
          isValid: false,
          error: 'Method not allowed by CORS policy'
        };
      }
 
      // Verificar headers
      if (request.headers && !this.areHeadersAllowed(request.headers)) {
        return {
          isValid: false,
          error: 'Headers not allowed by CORS policy'
        };
      }
 
      // Gerar headers para resposta válida
      const headers = this.generateCORSHeaders(request);
 
      return {
        isValid: true,
        headers: headers as unknown as Record<string, string>
      };
 
    } catch (error) {
      return {
        isValid: false,
        error: 'Internal CORS validation error'
      };
    }
  }
 
  // Gerar todos os headers de segurança para resposta
  public generateSecurityHeaders(request?: RequestInfo): Record<string, string> {
    const headers: Record<string, string> = {};
 
    // Adicionar headers CORS se necessário
    if (request) {
      const corsHeaders = this.generateCORSHeaders(request);
      Object.assign(headers, corsHeaders);
    }
 
    // Adicionar headers de segurança
    Object.assign(headers, this.securityHeaders);
 
    // Adicionar headers anti-cache para requisições sensíveis
    if (request?.method && ['POST', 'PUT', 'DELETE', 'PATCH'].includes(request.method.toUpperCase())) {
      Object.assign(headers, {
        'Cache-Control': 'no-store, no-cache, must-revalidate, proxy-revalidate',
        'Pragma': 'no-cache',
        'Expires': '0'
      });
    }
 
    return headers;
  }
 
  // Simular middleware CORS (para ambiente frontend)
  public corsMiddleware(request: RequestInfo): {
    shouldProceed: boolean;
    headers: Record<string, string>;
    error?: string;
  } {
    // Se for requisição preflight
    if (request.method?.toUpperCase() === 'OPTIONS') {
      const headers = this.generatePreflightHeaders(request);
 
      return {
        shouldProceed: false, // Não prosseguir com a requisição principal
        headers: headers as unknown as Record<string, string>
      };
    }
 
    // Validar requisição
    const validation = this.validateCORSRequest(request);
 
    if (!validation.isValid) {
      return {
        shouldProceed: false,
        headers: this.generateSecurityHeaders(request),
        error: validation.error
      };
    }
 
    return {
      shouldProceed: true,
      headers: this.generateSecurityHeaders(request)
    };
  }
 
  // Aplicar headers a uma requisição fetch
  public async fetchWithCORS(
    url: string,
    options: RequestInit = {},
    requestInfo: RequestInfo = {}
  ): Promise<Response> {
    // Simular origem do browser se não fornecida
    if (!requestInfo.origin && typeof window !== 'undefined') {
      requestInfo.origin = window.location.origin;
    }
 
    // Aplicar middleware CORS
    const middleware = this.corsMiddleware(requestInfo);
 
    if (!middleware.shouldProceed) {
      throw new ApiError(
        middleware.error || 'CORS policy violation',
        ApiErrorType.NETWORK_ERROR,
        'CORS_ERROR'
      );
    }
 
    // Adicionar headers à requisição
    const fetchOptions: RequestInit = {
      ...options,
      headers: {
        ...middleware.headers,
        ...options.headers
      }
    };
 
    // Fazer requisição
    const response = await fetch(url, fetchOptions);
 
    // Validar resposta CORS
    if (!response.ok && response.type === 'opaque') {
      throw new ApiError(
        'CORS request failed - opaque response',
        ApiErrorType.NETWORK_ERROR,
        'CORS_FAILURE'
      );
    }
 
    return response;
  }
 
  // Configurar dinamicamente origens permitidas
  public setAllowedOrigins(origins: string[]): void {
    this.config.allowedOrigins = origins;
  }
 
  // Adicionar origem permitida
  public addAllowedOrigin(origin: string): void {
    if (!this.config.allowedOrigins.includes(origin)) {
      this.config.allowedOrigins.push(origin);
    }
  }
 
  // Remover origem permitida
  public removeAllowedOrigin(origin: string): void {
    const index = this.config.allowedOrigins.indexOf(origin);
    if (index > -1) {
      this.config.allowedOrigins.splice(index, 1);
    }
  }
 
  // Configurar headers de segurança personalizados
  public setSecurityHeaders(headers: Record<string, string>): void {
    this.securityHeaders = { ...this.securityHeaders, ...headers };
  }
 
  // Habilitar/desabilitar modo de desenvolvimento
  public setDevelopmentMode(enabled: boolean): void {
    if (enabled) {
      // Em desenvolvimento, ser mais permissivo
      this.config.allowedOrigins = [];
      this.securityHeaders['Content-Security-Policy'] =
        "default-src *; script-src * 'unsafe-inline' 'unsafe-eval'; style-src * 'unsafe-inline'; img-src * data:; font-src * data:; connect-src *; frame-ancestors *;";
    } else {
      // Em produção, usar configurações restritivas
      this.config.allowedOrigins = DEFAULT_CORS_CONFIG.allowedOrigins;
      this.securityHeaders['Content-Security-Policy'] = SECURITY_HEADERS['Content-Security-Policy'];
    }
  }
 
  // Atualizar configuração
  public updateConfig(newConfig: Partial<CORSConfig>): void {
    this.config = { ...this.config, ...newConfig };
  }
 
  // Obter configuração atual
  public getConfig(): CORSConfig {
    return { ...this.config };
  }
 
  // Validar configuração
  public validateConfig(): { isValid: boolean; errors: string[] } {
    const errors: string[] = [];
 
    if (!Array.isArray(this.config.allowedOrigins)) {
      errors.push('allowedOrigins deve ser um array');
    }
 
    if (!Array.isArray(this.config.allowedMethods)) {
      errors.push('allowedMethods deve ser um array');
    }
 
    if (!Array.isArray(this.config.allowedHeaders)) {
      errors.push('allowedHeaders deve ser um array');
    }
 
    if (this.config.maxAge < 0) {
      errors.push('maxAge deve ser positivo');
    }
 
    return {
      isValid: errors.length === 0,
      errors
    };
  }
 
  // Resetar para configuração padrão
  public resetToDefaults(): void {
    this.config = { ...DEFAULT_CORS_CONFIG };
    this.securityHeaders = { ...SECURITY_HEADERS };
  }
}
 
// Instância global do serviço CORS
export const corsService = new CORSService();
 
// Funções de utilidade para uso em componentes
 
// Wrapper para fetch com CORS
export async function secureFetch(
  url: string,
  options: RequestInit = {},
  corsOptions: Partial<RequestInfo> = {}
): Promise<Response> {
  return corsService.fetchWithCORS(url, options, corsOptions);
}
 
// Criar instância personalizada de CORS service
export function createCORSService(config: Partial<CORSConfig>): CORSService {
  return new CORSService(config);
}
 
// Verificar se uma URL é segura (mesma origem ou permitida)
export function isSecureURL(url: string, currentOrigin?: string): boolean {
  try {
    const urlObj = new URL(url);
    const origin = currentOrigin || (typeof window !== 'undefined' ? window.location.origin : '');
 
    if (!origin) return false;
 
    const currentOriginObj = new URL(origin);
 
    // Mesma origem
    if (urlObj.origin === currentOriginObj.origin) {
      return true;
    }
 
    // Verificar se é HTTPS (exceto localhost)
    if (urlObj.protocol !== 'https:' && !urlObj.hostname.includes('localhost')) {
      return false;
    }
 
    // Verificar se origem é permitida
    return corsService.isOriginAllowed(urlObj.origin);
  } catch {
    return false;
  }
}
